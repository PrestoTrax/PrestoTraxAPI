API Modeling Storage Schema

Storage Schema - User
{
	"_id" : Inherent ID
	"Username" : {String, Min:5, Max:64, Required},
	"Email" : {Email, Min:5, Max:64, Required},
	"Password" : {String, Min:5, Max:64, Required},
	"Account Creation Date": {Date, Default: "CURRENT_DATE"}, 
	"Last Login" : {Date, Default: "CURRENT_DATE"},
	"Suspended" : {Bool, Default: True},
	"Devices Owned" : {Number, Min: 0, Max: 1024}
	"Owned Device IDs" : {Array: Strings of IDs, Length: 0-1024}	// IF ARRAYS ARE POSSIBLE // Fallback: Seperate Owned Device Schema?
	"" : {}
} 


Storage Schema - Device
{
	"_id" : Inherent ID
	"Device Name" : {String, Min:5, Max: 64, Default: "TRENDY_RANDOM_NAME_GEN"},
	"Device Owner" : {_id},
	"Reported Lost" : {Bool, Default: False},
	"Device Location" : {Array: String},  	// IF ARRAYS ARE POSSIBLE // Fallback: Seperate Owned Device Schema?
	
}

Storage Schema - Device Ping
{
	"_id" : Inherent ID
	"Device Owner" : {_id},
	"Device Location" : {String (Lat/Long), Max: 64},
	"Timestamp" : {Date, Default: "CURRENT_DATE"},
	"Moving" : {Bool},
}
